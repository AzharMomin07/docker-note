how to dockerize spring boot application 

1) create spring boot application with postgres database 
2) Create Dockerfile

-----

FROM openjdk:19


COPY target/one-to-many-0.0.1-SNAPSHOT.jar .

ENTRYPOINT ["java" , "-jar" ,"one-to-many-0.0.1-SNAPSHOT.jar"]

----

3) docker build -t image_name .

4) create network 
  --- docker network create network_name

5) create postgres container with network_name
  --- docker run --name postgres -p 5432:5432 -e POSTGRES_DB=name --network mynetwork -e POSTGRES_PASSWORD=mysecretpassword -d postgres

6) create container for your application with network_name and postgres configuration
  ---  docker run --name demo1 --network spring-data -e SPRING_DATASOURCE_URL=jdbc:postgresql://postgres_db:5432/one-to-many -e SPRING
_DATASOURCE_USERNAME=aalam -e spring.jpa.hibernate.ddl-auto=update -p 8080:8080 -e SPRING_DATASOURCE_PASSWORD=root demo
